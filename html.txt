<script>
const data = {
    "napoje": {
        products:{
            "Coca-Cola 0.5L": {price:6.50, subs:{"Freeway Cola 0.5L":2.50}},
            "Pepsi 0.5L": {price:6.00, subs:{"Hoop Cola 0.5L":3.20}},
            "Red Bull 250ml": {price:7.50, subs:{"Tiger 250ml":3.50}},
            "Żywiec Zdrój 0.5L": {price:2.80, subs:{"Woda Biedronka":1.20}}
        }
    },
    "jedzenie": {
        products:{
            "Kinder Bueno": {price:4.50, subs:{"Baton z Lidla":2.20}},
            "Czekolada Milka": {price:5.50, subs:{"Czekolada Fin Carre":3.20}}
        }
    },
    "kosmetyki": { products: { "Żel pod prysznic Markowy": {price:15, subs:{"Żel Cien":6}} } },
    "srodki czystosci": { products: { "Płyn do szyb": {price:12, subs:{"Płyn W5":5}} } }
};

const catDiv = document.getElementById("categories");
const app = document.getElementById("app");
const prodSel = document.getElementById("product");
let currentCat;

// Generowanie przycisków kategorii
for(let c in data){
    catDiv.innerHTML += `<button class="category-btn" onclick="openCat('${c}', this)">${c}</button>`;
}

function openCat(c, btn){
    currentCat = c;
    // UI updates
    document.querySelectorAll('.category-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    
    app.style.display = "block";
    document.getElementById("results").style.display = "none";
    document.getElementById("catTitle").innerText = c;
    
    // Załaduj produkty
    prodSel.innerHTML = "";
    for(let p in data[c].products){
        prodSel.innerHTML += `<option value="${p}">${p}</option>`;
    }
}

function resetResults() {
    document.getElementById("results").style.display = "none";
}

function calc(){
    const q = parseFloat(document.getElementById("qty").value);
    if(!q || q <= 0){ alert("Podaj ilość!"); return; }

    const pName = prodSel.value;
    const base = data[currentCat].products[pName];
    
    const monthlySpend = base.price * q;
    const yearlySpend = monthlySpend * 12;

    // Lewa strona wyników
    document.getElementById("mainResult").innerHTML = `
        <p>cena: <b>${base.price.toFixed(2)} zł</b></p>
        <p>miesięcznie wydajesz:<br><b>${monthlySpend.toFixed(2)} zł</b></p>
        <p>rocznie wydajesz:<br><b>${yearlySpend.toFixed(2)} zł</b></p>
    `;

    // Prawa strona wyników (substytuty)
    let subsHtml = "";
    for(let s in base.subs){
        const subPrice = base.subs[s];
        const mSave = (base.price - subPrice) * q;
        const ySave = mSave * 12;

        subsHtml += `
            <div class="sub-item">
                <small>nazwa substytutu:</small><br><b>${s}</b><br>
                <small>cena: ${subPrice.toFixed(2)} zł</small>
                <div class="save-row">
                    <div class="save-box">m-c oszczędzasz:<br><b>${mSave.toFixed(2)} zł</b></div>
                    <div class="save-box">rocznie oszczędzasz:<br><b>${ySave.toFixed(2)} zł</b></div>
                </div>
            </div>
        `;
    }
    document.getElementById("substitutesResult").innerHTML = subsHtml;
    document.getElementById("results").style.display = "block";
}
</script>
